void turnBackRightUntilContact(int speed){
    while(backLeft.Value() || !backRight.Value()){
        if (!backLeft.Value() && !backRight.Value()){
            rightWheel.SetPercent(-1 * speed);
            leftWheel.SetPercent(0);
        } else{
            rightWheel.SetPercent(0);
            leftWheel.SetPercent(-1 * speed);
        }


        
    }
    rightWheel.SetPercent(0);
}


void turnBackLeftUntilContact(int speed){
    while(backLeft.Value() || !backRight.Value()){
        if(backLeft.Value() && backRight.Value()){
            rightWheel.SetPercent(0);
            leftWheel.SetPercent(-1 * speed);
        } else {
            rightWheel.SetPercent(-1 * speed);
            leftWheel.SetPercent(0);
        }
    }
    rightWheel.SetPercent(0);
}

void moveForwardsUntilContact(int speed){
    while(frontLeft.Value() || frontRight.Value()){
        rightWheel.SetPercent(speed);
        leftWheel.SetPercent(speed);
    }
    leftWheel.SetPercent(0);
    rightWheel.SetPercent(0);
}

void moveBackwardsUntilContact(int speed){
    while(backLeft.Value() || !backRight.Value()){
        rightWheel.SetPercent(-1 * speed);
        leftWheel.SetPercent(speed);
    }
    leftWheel.SetPercent(0);
    rightWheel.SetPercent(0);
}
